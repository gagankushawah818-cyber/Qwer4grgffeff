  <div id="linkOutput"></div>
</div>

<script>
  function generateLink() {
    const username = document.getElementById("username").value.trim();
    const linkOutput = document.getElementById("linkOutput");

    if (username === "") {
      linkOutput.innerHTML = "Please enter your username.";
      return;
    }

    const baseUrl = "https://example.com/account/";
    const accountLink = baseUrl + encodeURIComponent(username);
    linkOutput.innerHTML = `Your account link: <a href="${accountLink}" target="_blank">${accountLink}</a>`;
  }
</script>
